# Runtime dependencies for the mcp-guardrails sidecar.
#
# Design principle: pin ONLY where stability or wire-compat matters, otherwise
# use lower bounds + upper major caps so the resolver can pick the newest
# compatible release.
#
# The ONNX PromptGuard model (gravitee-io/Llama-Prompt-Guard-2-86M-onnx) is
# public and non-gated — NO HF_TOKEN required.

# --- gRPC stack (wire-compat critical: stubs generated with grpcio 1.81.x) ---
grpcio>=1.81,<1.82
grpcio-health-checking>=1.81,<1.82
# grpcio-tools is a BUILD-TIME / DEV dependency only — NOT listed here.

# protobuf 5.x and 6.x are both runtime-compatible for our generated stubs.
protobuf>=5.29,<7

# --- Transformers + ONNX Runtime for PromptGuard-2-86M inference ---
# The ONNX model is public + non-gated, so no HF_TOKEN is needed. We load the
# .onnx graph with onnxruntime directly and tokenise with transformers
# (tokenizer-only). optimum is NOT used — it hard-requires torch (~750MB),
# which would defeat the ONNX path's size advantage. Truly torch-free.
transformers>=4.51.3
onnxruntime>=1.20.0

# --- OpenAI client for AgentAlignment LLM (second-stage, opt-in) ---
# Only needed when ENABLE_AGENT_ALIGNMENT=1. The LLM client reads the API key
# directly from LF_ALIGNMENT_API_KEY.
openai>=1.76.0

# --- HuggingFace Hub (transitive via transformers, but listed for clarity) ---
# Used by the Dockerfile models stage to snapshot_download the ONNX model.
huggingface-hub>=0.30.2

# --- hf_transfer: parallel chunked HF downloads (2-5x faster for the ~350MB
#     model). Enabled by HF_HUB_ENABLE_HF_TRANSFER=1 in the Dockerfile build
#     and the local download-model task. ---
hf_transfer>=0.1.6

# --- OpenTelemetry (optional at runtime; sidecar degrades to stdout audit
#     if these are absent). Lower-bound only; the 1.x line is stable. ---
opentelemetry-sdk>=1.28.0
opentelemetry-exporter-otlp>=1.28.0
